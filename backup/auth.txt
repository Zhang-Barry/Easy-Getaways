import axios from 'axios';

// Purpose: submit correct username/password, get a new JWT token.
export const login = (email, password) => async dispatch => {
    const config = {
        headers: {
            'Content-Type': 'application/json'
        }
    };

    const body = JSON.stringify({ username, password });

    try {
        const res = await axios.post(`${process.env.REACT_APP_API_URL}/dj-rest-auth/login/`, body, config);
        dispatch({
            type: LOGIN_SUCCESS,
            payload: res.data
        });
        return true;
    } catch (err) {
        dispatch({
            type: LOGIN_FAIL
        })
        return false;
    }
};

// Purpose: submit new user information AND login.
export const signup = (first_name, last_name, email, password, re_password) => async dispatch => {
    const config = {
        headers: {
            'Content-Type': 'application/json'
        }
    };

    const body = JSON.stringify({ first_name, last_name, email, password, re_password });

    try {
        const res = await axios.post(`${process.env.REACT_APP_API_URL}/dj-rest-auth/registration/`, body, config);
        dispatch({
            type: SIGNUP_SUCCESS,
            payload: res.data
        });
    } catch (err) {
        dispatch({
            type: SIGNUP_FAIL,
            payload: err.response.data
        })
    }
};

// delete user information and JWT stored locally.
export const logout = () => dispatch => {
    dispatch({
        type: LOGOUT
    });
};
